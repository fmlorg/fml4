#!/bin/sh
#reply-to と erros-to だけをつけるフィルター

sed '1,/^$/{
	   /^[Rr]eply-[Tt]o: /d
	   /^[Ee]rrors-[Tt]o: /d
	/^$/{c\
	Reply-To: ＭＬのアドレス\
	Errors-To: 管理者アドレス\

	}
}'
